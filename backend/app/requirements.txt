# ============================================
# 6. INSTALLATION STEPS
# ============================================
# 
# A. Flutter Frontend:
#    1. Cài đặt Flutter SDK
#    2. cd rubik (thư mục gốc dự án)
#    3. flutter pub get
#    4. flutter run
#
# B. Python Backend:
#    1. Cài đặt Python 3.8+
#    2. cd backend
#    3. pip install -r requirements.txt
#    4. Tạo file .env từ .env.example
#    5. Tạo database: mysql -u root -p < database_schema.sql
#    6. Chạy server: python -m app.main
#       hoặc: uvicorn app.main:app --reload
#
# C. Database Setup:
#    1. Cài đặt MySQL
#    2. Tạo database: CREATE DATABASE rubik_master;
#    3. Import schema: mysql -u root -p rubik_master < database_schema.sql

# 9. API ENDPOINTS (Backend)
# ============================================
# Authentication:
#   POST /api/auth/register
#   POST /api/auth/login
#
# Users:
#   GET /api/users/me
#   GET /api/users/online
#   GET /api/users/search/{username}
#
# Matches:
#   POST /api/matches/create
#   POST /api/matches/find-opponent
#   GET /api/matches/{match_id}
#   POST /api/matches/{match_id}/start
#   POST /api/matches/{match_id}/submit-result
#   GET /api/matches/
#
# Chat:
#   POST /api/chat/send
#   GET /api/chat/{match_id}/messages
#
# Friends:
#   POST /api/friends/request
#   POST /api/friends/{friendship_id}/accept
#   GET /api/friends/
#   GET /api/friends/pending
#
# WebSocket:
#   ws://localhost:8000/ws/{user_id}?token={access_token}
